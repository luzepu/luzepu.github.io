/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     struct ListNode *next;
 * };
 */


struct ListNode* addTwoNumbers(struct ListNode* l1, struct ListNode* l2){
    long long add1 = 0;
    long long add2 = 0;
    struct ListNode* temp;
		
    temp = l1;
    int count1 = 0;
    int c[100] = {0};
    while(temp!=NULL){
        c[count1] = temp->val;
        count1++;
        if(temp->next)
            temp = temp->next;
        else
            break;
    }
    while(count1--){
        add1 *= 10;
        add1 += c[count1]; 
    }
    
    temp = l2;
    int count2 = 0;
    int b[100] = {0};
    while(temp!=NULL){
        b[count2] = temp->val;
        count2++;
        if(temp->next)
            temp = temp->next;
        else
            break;
    }
    while(count2--){
        add2 *= 10;
        add2 += b[count2]; 
    }
    
    struct ListNode* L;
    L = (struct ListNode *)malloc(sizeof(struct ListNode));
    L->next = NULL;
    long long sum = add1 + add2;
    int a[100] = {0};
    int i=0;
    if(sum==0){
        i++;
    }
    while(sum!=0){ 
        a[i] = sum % 10;
        sum /= 10;
        i++;
    }
    
    int j=i-1;
    struct ListNode *p;
    while(i--){
        
        p = (struct ListNode *)malloc(sizeof(struct ListNode));
        p->next = L->next;
        p->val = a[j];
        L->next = p;
        j--;
    }
    
    return L->next;

}
